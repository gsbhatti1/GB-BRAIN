function drawLine(bars) {
    // Test using the spot exchange object to get K-line data. If testing with a futures exchange object, need to set the contract first.
    if (!bars) {
        return;
    }

    if (lastKLineTime == 0) { // First start-up, from the beginning

        begin = bars.length - 2; // Theoretically only need to look at the latest data point, but to prevent errors, check two points

        return;
    }

    let atr = talib.ATR(bars, atrNum); // Calculate average price amplitude

    for (let i = begin; i < bars.length; i++) {
        let bar = bars[i];
        if (bar.Time <= lastKLineTime) { // Already calculated and plotted
            continue;
        }

        let avg_pra = atr[i] / bar.Close;

        let hl2 = (bar.High + bar.Low) / 2;
        // Calculate grid
        let grid = hl2 * avg_pra / pp;
        let p1 = hl2 + grid * StopGrid;
        let p2 = hl2 + grid * 2;
        let p0 = hl2;
        let p3 = hl2 - grid * 2;
        let p4 = hl2 - grid * StopGrid;

        //------------Draw K-line chart, etc.

            PlotMultLine(cfgName1, "p3", p3, bar.Time, '#265ec5');
            PlotMultLine(cfgName1, "p4", p4, bar.Time, '#FF9800');
            continue;
        }

        //------------Make trading decisions based on the SMA value from the last data point (for loop only enters for the last time)

            return;
