// This source code is subject to the terms of the Mozilla Public License 2.0 at https://mozilla.org/MPL/2.0/
// © tbiktag

// Delta-RSI Oscillator Strategy With Filters
//
// This is a version of the Delta-RSI Oscillator Strategy compatible with 
// the Strategy Tester.
//
// This version also allows filtering the trade signals generated by Delts-RSI
// by means of volatility (defined by ATR), relative volume and RSI(14).
//
// Delta-RSI (© tbiktag) is a smoothed time derivative of the RSI designed
// as a momentum indicator. For the original publication, see link below:
// https://www.tradingview.com/script/OXQVFTQD-Delta-RSI-Oscillator/
// 
// D-RSI model parameters:
// RSI Length: The timeframe of the RSI that serves as an input to D-RSI.
// Frame Length: The length of the lookback frame used for local regression.
// Polynomial Order: The order of the local polynomial interpolation.

//@version=5
strategy("Polynomial-Interpolation-Based-RSI-Momentum-Strategy", overlay=true, default_qty_type=strategy.percent_of_equity, default_qty_value=10)

rsi_length = input(21, title="RSI Length")
window = input(50, title="Frame Length ( > Order)")
order = input(3, title="Polynomial Order")

// Calculate Delta-RSI
d_rsi = ta.rsi(rsi_length)
fit_coeff = array.new_float(window)
for i = 0 to window - 1
    fit_coeff.push(ta.polyval(order, d_rsi[i]))

dslope = na
if bar_index > (window + rsi_length) - 2
    dslope := ta.slope(fit_coeff, 1)

// Buy/Sell conditions
long_condition = crossover(dslope, 0)
short_condition = crossunder(dslope, 0)

// Filter conditions
atr_length = input(5, title="ATR Length")
atr = ta.atr(atr_length)
vol_threshold = input(30, title="Volume >")
rsi_hi = input(100, title="< RSI (14) >")

long_filter = v_input_10 and (not na(atr) and atr > v_input_18) and (ta.volume > vol_threshold) and (ta.rsi < rsi_hi)
short_filter = v_input_11 and (not na(atr) and atr > v_input_18) and (ta.volume > vol_threshold) and (ta.rsi < rsi_hi)

if long_condition and long_filter
    strategy.entry("Long", strategy.long)
    
if short_condition and short_filter
    strategy.entry("Short", strategy.short)

// Exit conditions
exit_condition = ta.crossover(dslope, 0) or ta.crossunder(dslope, 0)
if exit_condition and v_input_12
    strategy.close("Long")
    strategy.close("Short")

// Stop Loss/Take Profit
sl_percent = input(10, title="SL, %")
tp_percent = input(20, title="TP, %")

if long_filter and ta.barssince(long_condition) > 0 
    sl_price = strategy.position_avg_price * (1 - (sl_percent / 100))
    tp_price = strategy.position_avg_price * (1 + (tp_percent / 100))
    
// Add stop loss and take profit to the long position
strategy.exit("Long SL", "Long", limit=tp_price, stop=sl_price)

if short_filter and ta.barssince(short_condition) > 0 
    sl_price = strategy.position_avg_price * (1 + (sl_percent / 100))
    tp_price = strategy.position_avg_price * (1 - (tp_percent / 100))
    
// Add stop loss and take profit to the short position
strategy.exit("Short SL", "Short", limit=tp_price, stop=sl_price)
