//A very simple single-currency dynamic equilibrium strategy, buy when a certain proportion is lower than 50%, and sell when it is higher than 50%.

function main() {
    var STATE_IDLE = -1;
    var state = STATE_IDLE; //Indicates short position
    var entryPrice = 0;
    var initAccount = _C(exchange.GetAccount);
    var obj;
    var allAmount;
    var cashRatio;
    Log(initAccount);
    while (true) {
        var account = _C(exchange.GetAccount);
        var ticker = _C(exchange.GetTicker);
        if (state === STATE_IDLE) { //Initialization state is default position; default is short position, only buy
            obj = $.Buy(_N(account.Balance * 0.5 / ticker.Sell, 3));
            if (obj) { //If the purchase is successful, mark the opening of the position
                opAmount = obj.amount;
                entryPrice = obj.price;
                state = PD_LONG;
                account = _C(exchange.GetAccount);
                Log("Open a position to buy", obj.amount, "Price", obj.price, "Current number of coins held", account.Stocks);
            }
        } else { //state is not idle; handle dynamic balance detection
            allAmount = account.Balance + account.Stocks * ticker.Sell; //Calculate the total amount
            cashRatio = parseFloat((account.Balance / allAmount).toFixed(3));
            //Log("cash", account.Balance, "total assets", allAmount, "ratio", cashRatio);
            if (cashRatio > 0.5 + changeRatio) { //More cash, need to buy currency
                obj = $.Buy(_N(allAmount * (cashRatio - 0.5) / ticker.Sell / 2.0, 3)); //Buy the extra half to achieve balance
                if (obj) {
                    Log("Open a position to buy", obj.amount, "Price", obj.price);
                    Log("Current Funds", allAmount, "Profit", allAmount - initAccount.Balance);
                }
            } else if (cashRatio < 0.5 - changeRatio) { //Low cash, need to toss coins
                obj = $.Sell(_N(allAmount * (0.5 - cashRatio) / ticker.Sell / 2.0, 3)); //Buy the extra part
                if (obj) {
                    Log("Close the position and buy", obj.amount, "Price", obj.price);
                    Log("Current Funds", allAmount, "Profit", allAmount - initAccount.Balance);
                }
            }
        }
        Sleep(Interval * 1000);
    }
}
